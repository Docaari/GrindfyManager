JavaScript para Prioridades e Rebuys
javascript// ===== SISTEMA DE PRIORIDADES =====
function changePriority(element) {
  const priorities = [
    { class: 'priority-low', label: 'Baixa' },
    { class: 'priority-medium', label: 'Média' },
    { class: 'priority-high', label: 'Alta' }
  ];
  
  // Encontrar prioridade atual
  let currentIndex = priorities.findIndex(p => 
    element.classList.contains(p.class)
  );
  
  // Próxima prioridade (ciclo)
  let nextIndex = (currentIndex + 1) % priorities.length;
  let nextPriority = priorities[nextIndex];
  
  // Atualizar classes CSS
  priorities.forEach(p => element.classList.remove(p.class));
  element.classList.add(nextPriority.class);
  element.textContent = nextPriority.label;
  
  // Feedback visual
  element.style.transform = 'scale(1.2)';
  setTimeout(() => {
    element.style.transform = 'scale(1)';
  }, 200);
  
  // Salvar no estado
  const tournamentId = element.dataset.tournamentId;
  savePriority(tournamentId, nextIndex + 1);
  
  // Reordenar torneios
  sortTournamentsByPriority();
}

// ===== SISTEMA DE REBUYS =====
function addRebuy() {
  const tournamentCard = event.target.closest('.tournament-card');
  const rebuyCounter = tournamentCard.querySelector('.rebuy-counter');
  
  let currentRebuys = parseInt(rebuyCounter.dataset.rebuys) || 0;
  currentRebuys++;
  
  // Atualizar display
  rebuyCounter.dataset.rebuys = currentRebuys;
  rebuyCounter.textContent = `${currentRebuys} Rebuy${currentRebuys !== 1 ? 's' : ''}`;
  
  // Aplicar cores de alerta
  rebuyCounter.classList.remove('warning', 'danger');
  if (currentRebuys >= 4) {
    rebuyCounter.classList.add('danger');
  } else if (currentRebuys >= 2) {
    rebuyCounter.classList.add('warning');
  }
  
  // Feedback visual
  rebuyCounter.style.transform = 'scale(1.3)';
  setTimeout(() => {
    rebuyCounter.style.transform = 'scale(1)';
  }, 300);
  
  // Atualizar métricas
  updateMetrics();
}

function sortTournamentsByPriority() {
  const categories = document.querySelectorAll('.tournament-category');
  
  categories.forEach(category => {
    const cards = Array.from(category.querySelectorAll('.tournament-card'));
    
    cards.sort((a, b) => {
      const priorityA = getPriorityValue(a);
      const priorityB = getPriorityValue(b);
      const timeA = getTimeValue(a);
      const timeB = getTimeValue(b);
      
      // Ordenar por: Prioridade (desc) -> Horário (asc)
      if (priorityA !== priorityB) {
        return priorityB - priorityA;
      }
      return timeA - timeB;
    });
    
    // Reordenar no DOM
    cards.forEach(card => category.appendChild(card));
  });
}

function getPriorityValue(card) {
  const priorityBadge = card.querySelector('.priority-badge');
  if (!priorityBadge) return 0;
  
  if (priorityBadge.classList.contains('priority-high')) return 3;
  if (priorityBadge.classList.contains('priority-medium')) return 2;
  if (priorityBadge.classList.contains('priority-low')) return 1;
  return 0;
}

function getTimeValue(card) {
  const timeElement = card.querySelector('.tournament-time');
  const timeText = timeElement.textContent;
  
  if (timeText.includes(':')) {
    const [hours, minutes] = timeText.split(':').map(Number);
    return hours * 60 + minutes;
  }
  
  return 9999; // Para torneios "Em Andamento" ou "Finalizado"
}

function savePriority(tournamentId, priority) {
  // Implementar salvamento no backend
  fetch(`/api/tournaments/${tournamentId}`, {
    method: 'PATCH',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ priority })
  }).catch(console.error);
}
