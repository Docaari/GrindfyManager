JavaScript para Cálculo de Distribuições
javascript// Função para calcular distribuições
function calculateDistributions(sessionsData) {
  const distributions = {
    types: {},
    speeds: {}
  };

  let totalTournaments = 0;

  // Contar ocorrências
  sessionsData.forEach(session => {
    session.tournaments.forEach(tournament => {
      totalTournaments++;
      
      // Tipos
      const type = tournament.type.toLowerCase();
      distributions.types[type] = (distributions.types[type] || 0) + 1;
      
      // Velocidades
      const speed = tournament.speed.toLowerCase();
      distributions.speeds[speed] = (distributions.speeds[speed] || 0) + 1;
    });
  });

  // Calcular percentuais
  Object.keys(distributions.types).forEach(type => {
    const count = distributions.types[type];
    distributions.types[type] = {
      count,
      percentage: ((count / totalTournaments) * 100).toFixed(1)
    };
  });

  Object.keys(distributions.speeds).forEach(speed => {
    const count = distributions.speeds[speed];
    distributions.speeds[speed] = {
      count,
      percentage: ((count / totalTournaments) * 100).toFixed(1)
    };
  });

  return distributions;
}

// Função para atualizar interface
function updateDistributionUI(distributions) {
  // Atualizar tipos
  Object.keys(distributions.types).forEach(type => {
    const element = document.querySelector(`.type-${type}`);
    if (element) {
      const countElement = element.querySelector('.item-count');
      const percentageElement = element.querySelector('.item-percentage');
      
      const data = distributions.types[type];
      countElement.textContent = `${data.count} torneios`;
      percentageElement.textContent = `${data.percentage}%`;
    }
  });

  // Atualizar velocidades
  Object.keys(distributions.speeds).forEach(speed => {
    const element = document.querySelector(`.speed-${speed}`);
    if (element) {
      const countElement = element.querySelector('.item-count');
      const percentageElement = element.querySelector('.item-percentage');
      
      const data = distributions.speeds[speed];
      countElement.textContent = `${data.count} torneios`;
      percentageElement.textContent = `${data.percentage}%`;
    }
  });
}
