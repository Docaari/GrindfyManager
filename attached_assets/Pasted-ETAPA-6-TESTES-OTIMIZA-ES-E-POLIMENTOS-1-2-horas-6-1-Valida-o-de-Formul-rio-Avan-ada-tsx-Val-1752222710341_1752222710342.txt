6.1 Validação de Formulário Avançada
tsx// Validador de formulário completo
const createSessionValidator = () => {
  const schema = {
    volume: {
      required: false,
      min: 0,
      max: 100,
      message: "Volume deve estar entre 0 e 100"
    },
    profit: {
      required: false,
      message: "Profit deve ser um número válido"
    },
    abiMed: {
      required: false,
      min: 0,
      message: "ABI médio não pode ser negativo"
    },
    fts: {
      required: false,
      min: 0,
      validate: (value: number, data: EditSessionData) => 
        value <= data.volume || "FTs não pode ser maior que volume",
      message: "FTs inválido"
    },
    cravadas: {
      required: false,
      min: 0,
      validate: (value: number, data: EditSessionData) => 
        value <= data.fts || "Cravadas não pode ser maior que FTs",
      message: "Cravadas inválido"
    }
  };

  const validate = (data: EditSessionData): { isValid: boolean; errors: Record<string, string> } => {
    const errors: Record<string, string> = {};

    Object.entries(schema).forEach(([field, rules]) => {
      const value = data[field as keyof EditSessionData];
      
      if (rules.min !== undefined && typeof value === 'number' && value < rules.min) {
        errors[field] = rules.message;
      }
      
      if (rules.max !== undefined && typeof value === 'number' && value > rules.max) {
        errors[field] = rules.message;
      }
      
      if (rules.validate && typeof rules.validate === 'function') {
        const customValidation = rules.validate(value as number, data);
        if (customValidation !== true) {
          errors[field] = customValidation as string;
        }
      }
    });

    return {
      isValid: Object.keys(errors).length === 0,
      errors
    };
  };

  return { validate };
};
6.2 Auto-save e Recovery
tsx// Hook para auto-save
const useAutoSave = (editData: EditSessionData, sessionId: string) => {
  const [lastSaved, setLastSaved] = useState<Date | null>(null);
  const [hasUnsavedChanges, setHasUnsavedChanges] = useState(false);

  // Auto-save no localStorage a cada 30 segundos
  useEffect(() => {
    const autoSaveKey = `edit-session-${sessionId}`;
    
    const interval = setInterval(() => {
      if (hasUnsavedChanges) {
        localStorage.setItem(autoSaveKey, JSON.stringify({
          data: editData,
          timestamp: new Date().toISOString()
        }));
        setLastSaved(new Date());
        setHasUnsavedChanges(false);
      }
    }, 30000);

    return () => clearInterval(interval);
  }, [editData, sessionId, hasUnsavedChanges]);

  // Detectar mudanças
  useEffect(() => {
    setHasUnsavedChanges(true);
  }, [editData]);

  // Limpar auto-save ao salvar com sucesso
  const clearAutoSave = () => {
    localStorage.removeItem(`edit-session-${sessionId}`);
    setHasUnsavedChanges(false);
    setLastSaved(new Date());
  };

  // Recuperar dados salvos
  const recoverAutoSave = (): EditSessionData | null => {
    const autoSaveKey = `edit-session-${sessionId}`;
    const saved = localStorage.getItem(autoSaveKey);
    
    if (saved) {
      try {
        const { data, timestamp } = JSON.parse(saved);
        const saveTime = new Date(timestamp);
        const now = new Date();
        
        // Só recuperar se foi salvo nas últimas 2 horas
        if (now.getTime() - saveTime.getTime() < 2 * 60 * 60 * 1000) {
          return data;
        }
      } catch (error) {
        console.error('Erro ao recuperar auto-save:', error);
      }
    }
    
    return null;
  };

  return {
    lastSaved,
    hasUnsavedChanges,
    clearAutoSave,
    recoverAutoSave
  };
};
6.3 Testes de Acessibilidade
tsx// Adicionar atributos de acessibilidade
const AccessibleModal = () => {
  const modalRef = useRef<HTMLDivElement>(null);
  
  // Focus trap
  useEffect(() => {
    const modal = modalRef.current;
    if (!modal) return;

    const focusableElements = modal.querySelectorAll(
      'button, [href], input, select, textarea, [tabindex]:not([tabindex="-1"])'
    );
    
    const firstFocusable = focusableElements[0] as HTMLElement;
    const lastFocusable = focusableElements[focusableElements.length - 1] as HTMLElement;

    const handleTabKey = (e: KeyboardEvent) => {
      if (e.key === 'Tab') {
        if (e.shiftKey) {
          if (document.activeElement === firstFocusable) {
            lastFocusable.focus();
            e.preventDefault();
          }
        } else {
          if (document.activeElement === lastFocusable) {
            firstFocusable.focus();
            e.preventDefault();
          }
        }
      }
    };

    modal.addEventListener('keydown', handleTabKey);
    firstFocusable?.focus();

    return () => modal.removeEventListener('keydown', handleTabKey);
  }, []);

  return (
    <div
      ref={modalRef}
      role="dialog"
      aria-labelledby="modal-title"
      aria-describedby="modal-description"
      aria-modal="true"
    >
      {/* Conteúdo do modal */}
    </div>
  );
};

// Debounce para validação
const useDebouncedValidation = (editData: EditSessionData, delay = 500) => {
  const [errors, setErrors] = useState<Record<string, string>>({});
  
  const debouncedValidate = useMemo(
    () => debounce((data: EditSessionData) => {
      const { errors: validationErrors } = createSessionValidator().validate(data);
      setErrors(validationErrors);
    }, delay),
    [delay]
  );

  useEffect(() => {
    debouncedValidate(editData);
  }, [editData, debouncedValidate]);

  return errors;
};
