Função para calcular dados do sparkline:
jsxconst generateSparklineData = (metric, period = 90) => {
  // USAR dados existentes para gerar array de valores dos últimos 90 dias
  // Agrupar por semana para ter ~12-13 pontos no gráfico
  const weeklyData = groupDataByWeeks(metric, period);
  
  return weeklyData.map((week, index) => ({
    week: index + 1,
    value: week[metric]
  }));
};

const calculateTrend = (current, previous) => {
  if (!previous || previous === 0) return 0;
  return ((current - previous) / previous) * 100;
};

Função para calcular comparações de período:
jsxconst calculatePeriodComparison = (currentPeriod, previousPeriod) => {
  return {
    profit: {
      current: currentPeriod.totalProfit,
      previous: previousPeriod.totalProfit,
      change: ((currentPeriod.totalProfit - previousPeriod.totalProfit) / previousPeriod.totalProfit) * 100
    },
    roi: {
      current: currentPeriod.roi,
      previous: previousPeriod.roi,
      change: currentPeriod.roi - previousPeriod.roi
    },
    volume: {
      current: currentPeriod.tournamentCount,
      previous: previousPeriod.tournamentCount,
      change: ((currentPeriod.tournamentCount - previousPeriod.tournamentCount) / previousPeriod.tournamentCount) * 100
    },
    bestSite: findMostProfitableSite(currentPeriod.siteData)
  };
};

jsxconst generateVelocityData = (tournaments, metric) => {
  const velocityGroups = tournaments.reduce((acc, tournament) => {
    const velocity = tournament.speed || 'Normal';
    if (!acc[velocity]) {
      acc[velocity] = { tournaments: [], totalProfit: 0, totalVolume: 0 };
    }
    acc[velocity].tournaments.push(tournament);
    acc[velocity].totalProfit += tournament.profit || 0;
    acc[velocity].totalVolume += 1;
    return acc;
  }, {});
  
  return Object.entries(velocityGroups).map(([velocity, data]) => ({
    velocity,
    volume: data.totalVolume,
    profit: data.totalProfit,
    avgProfit: data.totalProfit / data.totalVolume
  }));
};
