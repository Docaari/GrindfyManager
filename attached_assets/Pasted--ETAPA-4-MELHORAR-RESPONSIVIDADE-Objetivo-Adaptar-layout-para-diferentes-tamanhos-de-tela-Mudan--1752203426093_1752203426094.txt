JavaScript para Detecção de Dispositivo
javascript// Detectar tamanho da tela e ajustar layout
function adjustLayoutForScreen() {
  const screenWidth = window.innerWidth;
  const daysGrid = document.querySelector('.days-grid');
  
  if (screenWidth < 768) {
    // Mobile: considerar scroll horizontal
    daysGrid.classList.add('days-grid-horizontal');
    daysGrid.classList.remove('days-grid');
  } else {
    // Desktop/Tablet: grid normal
    daysGrid.classList.add('days-grid');
    daysGrid.classList.remove('days-grid-horizontal');
  }
}

// Executar no carregamento e redimensionamento
window.addEventListener('load', adjustLayoutForScreen);
window.addEventListener('resize', adjustLayoutForScreen);
