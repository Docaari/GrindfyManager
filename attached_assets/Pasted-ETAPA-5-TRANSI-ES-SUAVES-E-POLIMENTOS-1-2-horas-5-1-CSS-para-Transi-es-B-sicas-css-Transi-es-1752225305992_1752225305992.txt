5.2 Indicadores Visuais de Estado
tsxconst StatusIndicator = ({ 
  preparationPercentage, 
  screenCap, 
  hasErrors 
}: {
  preparationPercentage: number;
  screenCap: number;
  hasErrors: boolean;
}) => {
  const getStatusColor = () => {
    if (hasErrors) return 'text-red-400';
    if (preparationPercentage < 40) return 'text-yellow-400';
    if (preparationPercentage >= 80) return 'text-green-400';
    return 'text-blue-400';
  };

  const getStatusText = () => {
    if (hasErrors) return 'Configuração incompleta';
    if (preparationPercentage < 40) return 'Preparação baixa';
    if (preparationPercentage >= 80) return 'Muito bem preparado';
    return 'Configuração ok';
  };

  return (
    <div className={`status-indicator ${getStatusColor()}`}>
      <div className="flex items-center gap-2">
        <div className="status-dot"></div>
        <span className="text-sm">{getStatusText()}</span>
      </div>
    </div>
  );
};
