generator client {
  provider = "prisma-client-js"
}

datasource db {
  provider = "postgresql"
  url      = env("DATABASE_URL")
}

model User {
  id               String   @id @default(cuid())
  email            String   @unique
  passwordHash     String   @map("password_hash")
  createdAt        DateTime @default(now()) @map("created_at")
  subscriptionType String   @default("free") @map("subscription_type")
  
  // Configurações do usuário
  timezone         String   @default("America/Sao_Paulo")
  currency         String   @default("BRL")
  
  // Relacionamentos
  tournaments       Tournament[]
  tournamentTemplates TournamentTemplate[]
  weeklyPlans       WeeklyPlan[]
  grindSessions     GrindSession[]
  preparationLogs   PreparationLog[]
  customGroups      CustomGroup[]
  
  @@map("users")
}

model Tournament {
  id          String   @id @default(cuid())
  userId      String   @map("user_id")
  name        String
  buyIn       Decimal  @map("buy_in")
  prizePool   Decimal? @map("prize_pool")
  position    Int?
  prize       Decimal  @default(0)
  datePlayed  DateTime @map("date_played")
  site        String
  format      String   // MTT, SNG, etc
  category    String   // Vanilla, PKO, Mystery, etc
  speed       String   // Regular, Turbo, Hyper, etc
  
  // Campos adicionais para análise
  fieldSize       Int?     @map("field_size")
  reentries       Int      @default(0)
  finalTable      Boolean  @default(false) @map("final_table")
  bigHit          Boolean  @default(false) @map("big_hit") // Prêmio > 10x buy-in
  earlyFinish     Boolean  @default(false) @map("early_finish") // < 30% do field
  lateFinish      Boolean  @default(false) @map("late_finish") // > 70% do field
  currency        String   @default("BRL")
  
  // Timestamps
  createdAt   DateTime @default(now()) @map("created_at")
  updatedAt   DateTime @updatedAt @map("updated_at")
  
  // Relacionamentos
  user            User                @relation(fields: [userId], references: [id], onDelete: Cascade)
  template        TournamentTemplate? @relation(fields: [templateId], references: [id])
  templateId      String?             @map("template_id")
  grindSession    GrindSession?       @relation(fields: [grindSessionId], references: [id])
  grindSessionId  String?             @map("grind_session_id")
  
  @@map("tournaments")
}

model TournamentTemplate {
  id            String   @id @default(cuid())
  userId        String   @map("user_id")
  name          String
  site          String
  format        String
  category      String
  speed         String
  
  // Horários típicos
  dayOfWeek     Int[]    @map("day_of_week") // Array: 0=domingo, 1=segunda, etc
  startTime     String[] @map("start_time")  // Array de horários: ["18:00", "20:00"]
  
  // Estatísticas calculadas
  avgBuyIn      Decimal  @default(0) @map("avg_buyin")
  avgRoi        Decimal  @default(0) @map("avg_roi")
  totalPlayed   Int      @default(0) @map("total_played")
  totalProfit   Decimal  @default(0) @map("total_profit")
  finalTables   Int      @default(0) @map("final_tables")
  bigHits       Int      @default(0) @map("big_hits")
  avgFieldSize  Int?     @map("avg_field_size")
  
  // Timestamps
  createdAt     DateTime @default(now()) @map("created_at")
  updatedAt     DateTime @updatedAt @map("updated_at")
  lastPlayed    DateTime? @map("last_played")
  
  // Relacionamentos
  user              User                @relation(fields: [userId], references: [id], onDelete: Cascade)
  tournaments       Tournament[]
  plannedTournaments PlannedTournament[]
  customGroupTemplates CustomGroupTemplate[]
  
  @@map("tournament_templates")
}

model WeeklyPlan {
  id          String   @id @default(cuid())
  userId      String   @map("user_id")
  weekStart   DateTime @map("week_start") // Data de início da semana (segunda-feira)
  title       String?
  description String?
  
  // Metas da semana
  targetBuyins    Decimal? @map("target_buyins")
  targetProfit    Decimal? @map("target_profit")
  targetVolume    Int?     @map("target_volume") // Número de torneios
  
  // Status
  isActive    Boolean  @default(true) @map("is_active")
  
  // Timestamps
  createdAt   DateTime @default(now()) @map("created_at")
  updatedAt   DateTime @updatedAt @map("updated_at")
  
  // Relacionamentos
  user                User                @relation(fields: [userId], references: [id], onDelete: Cascade)
  plannedTournaments  PlannedTournament[]
  
  @@unique([userId, weekStart])
  @@map("weekly_plans")
}

model PlannedTournament {
  id          String   @id @default(cuid())
  planId      String   @map("plan_id")
  templateId  String   @map("template_id")
  dayOfWeek   Int      @map("day_of_week") // 0=domingo, 1=segunda, etc
  startTime   String   @map("start_time")  // "18:00"
  
  // Status
  isPlayed    Boolean  @default(false) @map("is_played")
  notes       String?
  
  // Timestamps
  createdAt   DateTime @default(now()) @map("created_at")
  
  // Relacionamentos
  plan        WeeklyPlan         @relation(fields: [planId], references: [id], onDelete: Cascade)
  template    TournamentTemplate @relation(fields: [templateId], references: [id], onDelete: Cascade)
  
  @@map("planned_tournaments")
}

model GrindSession {
  id              String   @id @default(cuid())
  userId          String   @map("user_id")
  date            DateTime
  
  // Planejamento vs Realidade
  plannedBuyins   Decimal  @default(0) @map("planned_buyins")
  actualBuyins    Decimal  @default(0) @map("actual_buyins")
  profitLoss      Decimal  @default(0) @map("profit_loss")
  
  // Controle de tempo
  duration        Int?     // Duração em minutos
  startTime       DateTime? @map("start_time")
  endTime         DateTime? @map("end_time")
  
  // Status da sessão
  status          String   @default("planned") // planned, active, completed, cancelled
  
  // Métricas da sessão
  tournamentsPlayed   Int     @default(0) @map("tournaments_played")
  finalTables         Int     @default(0) @map("final_tables")
  bigHits             Int     @default(0) @map("big_hits")
  
  // Observações
  notes           String?
  
  // Timestamps
  createdAt       DateTime @default(now()) @map("created_at")
  updatedAt       DateTime @updatedAt @map("updated_at")
  
  // Relacionamentos
  user            User             @relation(fields: [userId], references: [id], onDelete: Cascade)
  tournaments     Tournament[]
  preparationLogs PreparationLog[]
  
  @@map("grind_sessions")
}

model PreparationLog {
  id                  String   @id @default(cuid())
  userId              String   @map("user_id")
  sessionId           String?  @map("session_id")
  
  // Estado mental (1-5)
  mentalState         Int      @map("mental_state")
  focusLevel          Int      @map("focus_level")
  confidenceLevel     Int      @map("confidence_level")
  
  // Exercícios completados
  exercisesCompleted  String[] @map("exercises_completed") // Array de exercícios
  warmupCompleted     Boolean  @default(false) @map("warmup_completed")
  
  // Metas e notas
  sessionGoals        String?  @map("session_goals")
  notes               String?
  
  // Review pós-sessão
  postSessionReview   String?  @map("post_session_review")
  goalsAchieved       Boolean? @map("goals_achieved")
  lessonsLearned      String?  @map("lessons_learned")
  
  // Timestamps
  createdAt           DateTime @default(now()) @map("created_at")
  
  // Relacionamentos
  user                User          @relation(fields: [userId], references: [id], onDelete: Cascade)
  session             GrindSession? @relation(fields: [sessionId], references: [id], onDelete: SetNull)
  
  @@map("preparation_logs")
}

model CustomGroup {
  id          String   @id @default(cuid())
  userId      String   @map("user_id")
  name        String
  description String?
  color       String?  // Hex color para UI
  
  // Critérios do grupo
  criteria    Json     // Flexível para diferentes tipos de filtros
  
  // Timestamps
  createdAt   DateTime @default(now()) @map("created_at")
  updatedAt   DateTime @updatedAt @map("updated_at")
  
  // Relacionamentos
  user        User                   @relation(fields: [userId], references: [id], onDelete: Cascade)
  templates   CustomGroupTemplate[]
  
  @@map("custom_groups")
}

model CustomGroupTemplate {
  id         String @id @default(cuid())
  groupId    String @map("group_id")
  templateId String @map("template_id")
  
  // Relacionamentos
  group      CustomGroup        @relation(fields: [groupId], references: [id], onDelete: Cascade)
  template   TournamentTemplate @relation(fields: [templateId], references: [id], onDelete: Cascade)
  
  @@unique([groupId, templateId])
  @@map("custom_group_templates")
}

// Tabela para armazenar insights e sugestões do Grade Coach
model CoachingInsight {
  id          String   @id @default(cuid())
  userId      String   @map("user_id")
  type        String   // "suggestion", "warning", "opportunity"
  category    String   // "roi_optimization", "volume_adjustment", "schedule_conflict"
  title       String
  description String
  priority    Int      @default(1) // 1=baixa, 2=média, 3=alta
  
  // Dados do insight
  data        Json?    // Dados específicos do insight
  
  // Status
  isRead      Boolean  @default(false) @map("is_read")
  isApplied   Boolean  @default(false) @map("is_applied")
  
  // Timestamps
  createdAt   DateTime @default(now()) @map("created_at")
  expiresAt   DateTime? @map("expires_at")
  
  @@map("coaching_insights")
}

// Tabela para configurações do usuário
model UserSettings {
  id                    String  @id @default(cuid())
  userId                String  @unique @map("user_id")
  
  // Configurações de análise
  bigHitMultiplier      Decimal @default(10) @map("big_hit_multiplier") // 10x buy-in
  earlyFinishThreshold  Decimal @default(0.3) @map("early_finish_threshold") // 30%
  lateFinishThreshold   Decimal @default(0.7) @map("late_finish_threshold") // 70%
  
  // Configurações de notificação
  emailNotifications    Boolean @default(true) @map("email_notifications")
  coachingAlerts        Boolean @default(true) @map("coaching_alerts")
  sessionReminders      Boolean @default(true) @map("session_reminders")
  
  // Configurações de exibição
  defaultChartPeriod    String  @default("30d") @map("default_chart_period")
  preferredCurrency     String  @default("BRL") @map("preferred_currency")
  darkMode              Boolean @default(false) @map("dark_mode")
  
  // Timestamps
  createdAt             DateTime @default(now()) @map("created_at")
  updatedAt             DateTime @updatedAt @map("updated_at")
  
  @@map("user_settings")
}
