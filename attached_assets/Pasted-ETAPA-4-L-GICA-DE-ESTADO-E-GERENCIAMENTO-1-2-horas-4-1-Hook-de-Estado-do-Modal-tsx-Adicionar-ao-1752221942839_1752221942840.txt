4.1 Hook de Estado do Modal
tsx// Adicionar ao GrindSession.tsx
const useEditSessionModal = () => {
  const [editData, setEditData] = useState<EditSessionData>({
    volume: 0,
    profit: 0,
    abiMed: 0,
    roi: 0,
    fts: 0,
    cravadas: 0,
    energiaMedia: 5,
    focoMedio: 5,
    confiancaMedia: 5,
    inteligenciaEmocionalMedia: 5,
    interferenciasMedia: 5,
    preparationNotes: '',
    dailyGoals: '',
    finalNotes: '',
    objectiveCompleted: false
  });

  const [validationErrors, setValidationErrors] = useState<Record<string, string>>({});

  const updateField = (field: keyof EditSessionData, value: any) => {
    setEditData(prev => ({ ...prev, [field]: value }));
    
    // Limpar erro do campo quando usuário edita
    if (validationErrors[field]) {
      setValidationErrors(prev => {
        const newErrors = { ...prev };
        delete newErrors[field];
        return newErrors;
      });
    }
  };

  const loadSessionData = (session: SessionHistoryData) => {
    setEditData({
      volume: session.volume || 0,
      profit: session.profit || 0,
      abiMed: session.abiMed || 0,
      roi: session.roi || 0,
      fts: session.fts || 0,
      cravadas: session.cravadas || 0,
      energiaMedia: session.energiaMedia || 5,
      focoMedio: session.focoMedio || 5,
      confiancaMedia: session.confiancaMedia || 5,
      inteligenciaEmocionalMedia: session.inteligenciaEmocionalMedia || 5,
      interferenciasMedia: session.interferenciasMedia || 5,
      preparationNotes: session.preparationNotes || '',
      dailyGoals: session.dailyGoals || '',
      finalNotes: session.finalNotes || '',
      objectiveCompleted: session.objectiveCompleted || false
    });
    setValidationErrors({});
  };

  const validateData = (): boolean => {
    const errors: Record<string, string> = {};

    // Validações de métricas
    if (editData.volume < 0) {
      errors.volume = 'Volume não pode ser negativo';
    }
    
    if (editData.fts > editData.volume) {
      errors.fts = 'FTs não pode ser maior que o volume';
    }
    
    if (editData.cravadas > editData.fts) {
      errors.cravadas = 'Cravadas não pode ser maior que FTs';
    }

    if (editData.abiMed < 0) {
      errors.abiMed = 'ABI médio não pode ser negativo';
    }

    // Validações de estado mental
    const mentalFields = ['energiaMedia', 'focoMedio', 'confiancaMedia', 'inteligenciaEmocionalMedia', 'interferenciasMedia'];
    mentalFields.forEach(field => {
      const value = editData[field as keyof EditSessionData] as number;
      if (value < 0 || value > 10) {
        errors[field] = 'Valor deve estar entre 0 e 10';
      }
    });

    setValidationErrors(errors);
    return Object.keys(errors).length === 0;
  };

  return {
    editData,
    updateField,
    loadSessionData,
    validateData,
    validationErrors
  };
};
4.2 Integração com React Query
tsx// Atualizar mutation existente no GrindSession.tsx
const editSessionMutation = useMutation({
  mutationFn: async (data: { id: string; sessionData: EditSessionData }) => {
    return apiRequest(`/api/grind-sessions/${data.id}`, {
      method: "PUT",
      body: JSON.stringify(data.sessionData),
    });
  },
  onSuccess: () => {
    toast({
      title: "✅ Sessão atualizada!",
      description: "As informações da sessão foram atualizadas com sucesso.",
      duration: 3000,
    });
    setIsEditDialogOpen(false);
    setEditingSession(null);
    queryClient.invalidateQueries({ queryKey: ["/api/grind-sessions/history"] });
  },
  onError: (error: any) => {
    toast({
      title: "❌ Erro ao atualizar",
      description: error.message || "Algo deu errado ao salvar as alterações.",
      variant: "destructive",
      duration: 5000,
    });
  },
});
4.3 Handler de Salvamento
tsx// Adicionar ao GrindSession.tsx
const handleSaveEdit = async () => {
  if (!editingSession) return;
  
  // Validar dados
  if (!validateData()) {
    toast({
      title: "⚠️ Dados inválidos",
      description: "Por favor, corrija os erros antes de salvar.",
      variant: "destructive",
    });
    return;
  }

  // Calcular ROI automaticamente se profit e ABI estão preenchidos
  const calculatedROI = editData.abiMed > 0 && editData.volume > 0 
    ? ((editData.profit / (editData.abiMed * editData.volume)) * 100)
    : editData.roi;

  const finalData = {
    ...editData,
    roi: calculatedROI
  };

  editSessionMutation.mutate({
    id: editingSession.id,
    sessionData: finalData
  });
};
