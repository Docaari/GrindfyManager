JavaScript para Timer
javascript// ===== TIMER DE SESSÃO COM MENSAGENS MOTIVACIONAIS =====
let sessionStartTime = Date.now();

function updateSessionTimer() {
  const timer = document.getElementById('sessionTimer');
  const statusMessage = document.getElementById('statusMessage');
  
  setInterval(() => {
    const elapsed = Date.now() - sessionStartTime;
    const hours = Math.floor(elapsed / (1000 * 60 * 60));
    const minutes = Math.floor((elapsed % (1000 * 60 * 60)) / (1000 * 60));
    
    timer.textContent = `${hours}h ${minutes}m`;
    
    // Mensagens motivacionais baseadas no tempo
    if (hours === 0 && minutes < 30) {
      statusMessage.textContent = "🔥 Começando com tudo!";
    } else if (hours < 2) {
      statusMessage.textContent = "💪 Mantendo o foco!";
    } else if (hours < 4) {
      statusMessage.textContent = "🎯 No ritmo certo!";
    } else if (hours < 6) {
      statusMessage.textContent = "🏆 Maratona épica!";
    } else {
      statusMessage.textContent = "🚀 Sessão lendária!";
    }
  }, 60000); // Atualizar a cada minuto
}

// Função para inicializar sessão
function initializeSession() {
  sessionStartTime = Date.now();
  
  // Salvar no sessionStorage para persistir em caso de reload
  sessionStorage.setItem('sessionStartTime', sessionStartTime);
  
  updateSessionTimer();
}

// Função para recuperar sessão em caso de reload
function restoreSession() {
  const savedStartTime = sessionStorage.getItem('sessionStartTime');
  if (savedStartTime) {
    sessionStartTime = parseInt(savedStartTime);
  }
}

// Executar na inicialização
document.addEventListener('DOMContentLoaded', function() {
  restoreSession();
  updateSessionTimer();
});
