JavaScript para Cálculo de Trends
javascript// Função para calcular trends das métricas
function calculateMetricTrends(currentData, previousData) {
  const trends = {};
  
  const metrics = ['volume', 'profit', 'abi', 'roi', 'fts', 'wins'];
  
  metrics.forEach(metric => {
    const current = currentData[metric] || 0;
    const previous = previousData[metric] || 0;
    
    if (previous === 0) {
      trends[metric] = { percentage: 0, direction: 'neutral' };
      return;
    }
    
    const percentage = ((current - previous) / previous) * 100;
    const direction = percentage > 0 ? 'up' : percentage < 0 ? 'down' : 'neutral';
    
    trends[metric] = {
      percentage: Math.abs(percentage).toFixed(1),
      direction
    };
  });
  
  return trends;
}

// Função para aplicar trends nos cards
function applyTrendsToCards(trends) {
  Object.keys(trends).forEach(metric => {
    const card = document.querySelector(`.metric-${metric}`);
    const trendElement = card.querySelector('.metric-trend');
    
    const { percentage, direction } = trends[metric];
    
    trendElement.className = `metric-trend trend-${direction}`;
    trendElement.textContent = direction === 'up' ? `+${percentage}%` : 
                              direction === 'down' ? `-${percentage}%` : '0%';
  });
}

// Exemplo de uso
const currentPeriod = {
  volume: 10,
  profit: 40.70,
  abi: 56.92,
  roi: -15.9,
  fts: 0,
  wins: 0
};

const previousPeriod = {
  volume: 8,
  profit: 37.50,
  abi: 59.92,
  roi: -12.9,
  fts: 0,
  wins: 0
};

const trends = calculateMetricTrends(currentPeriod, previousPeriod);
applyTrendsToCards(trends);
