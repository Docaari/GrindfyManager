javascript// ===== GERENCIAMENTO DE CATEGORIAS =====
const tournamentCategories = {
  upcoming: [],
  active: [],
  finished: []
};

function moveTournamentToCategory(tournamentId, fromCategory, toCategory) {
  const tournament = tournamentCategories[fromCategory].find(t => t.id === tournamentId);
  
  if (tournament) {
    // Remover da categoria atual
    tournamentCategories[fromCategory] = tournamentCategories[fromCategory].filter(t => t.id !== tournamentId);
    
    // Adicionar na nova categoria
    tournament.status = toCategory;
    tournamentCategories[toCategory].push(tournament);
    
    // Atualizar interface
    updateCategoryUI(fromCategory);
    updateCategoryUI(toCategory);
    updateCategoryCounts();
  }
}

function updateCategoryUI(category) {
  const listElement = document.getElementById(`${category}List`);
  listElement.innerHTML = '';
  
  tournamentCategories[category].forEach(tournament => {
    const tournamentElement = createTournamentElement(tournament);
    listElement.appendChild(tournamentElement);
  });
}

function updateCategoryCounts() {
  document.getElementById('upcomingCount').textContent = tournamentCategories.upcoming.length;
  document.getElementById('activeCount').textContent = tournamentCategories.active.length;
  document.getElementById('finishedCount').textContent = tournamentCategories.finished.length;
}

function createTournamentElement(tournament) {
  const element = document.createElement('div');
  element.className = `tournament-card tournament-${tournament.status}`;
  element.innerHTML = generateTournamentHTML(tournament);
  return element;
}
