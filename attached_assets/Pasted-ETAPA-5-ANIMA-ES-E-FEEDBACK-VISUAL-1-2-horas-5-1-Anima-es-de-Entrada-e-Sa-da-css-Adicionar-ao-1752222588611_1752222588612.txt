5.2 Estados de Loading e Sucesso
tsx// Componente LoadingButton
const LoadingButton = ({ 
  isLoading, 
  children, 
  loadingText = "Salvando...",
  successText = "Salvo!",
  showSuccess = false,
  ...props 
}: ButtonProps & {
  isLoading: boolean;
  loadingText?: string;
  successText?: string;
  showSuccess?: boolean;
}) => {
  return (
    <Button 
      disabled={isLoading}
      className={cn(
        "transition-all duration-300",
        showSuccess && "bg-green-500 hover:bg-green-500"
      )}
      {...props}
    >
      {isLoading ? (
        <>
          <div className="animate-spin rounded-full h-4 w-4 border-b-2 border-white mr-2" />
          {loadingText}
        </>
      ) : showSuccess ? (
        <>
          <Check className="w-4 h-4 mr-2" />
          {successText}
        </>
      ) : (
        children
      )}
    </Button>
  );
};
5.3 Feedback Visual em Tempo Real
tsx// Hook para feedback visual
const useVisualFeedback = () => {
  const [savedField, setSavedField] = useState<string | null>(null);
  const [fieldErrors, setFieldErrors] = useState<Record<string, boolean>>({});

  const showFieldSaved = (fieldName: string) => {
    setSavedField(fieldName);
    setTimeout(() => setSavedField(null), 2000);
  };

  const setFieldError = (fieldName: string, hasError: boolean) => {
    setFieldErrors(prev => ({ ...prev, [fieldName]: hasError }));
  };

  const getFieldClassName = (fieldName: string, baseClass: string) => {
    return cn(
      baseClass,
      fieldErrors[fieldName] && "border-red-400 bg-red-400/10",
      savedField === fieldName && "border-green-400 bg-green-400/10",
      "transition-all duration-300"
    );
  };

  return {
    showFieldSaved,
    setFieldError,
    getFieldClassName,
    savedField,
    fieldErrors
  };
};
