JavaScript para Toggle
javascript// ===== TOGGLE DASHBOARD =====
let isDashboardVisible = true;

function toggleDashboard() {
  const content = document.getElementById('dashboardContent');
  const toggle = document.getElementById('dashboardToggle');
  
  isDashboardVisible = !isDashboardVisible;
  content.classList.toggle('collapsed');
  toggle.classList.toggle('collapsed');
  
  // Salvar preferência
  localStorage.setItem('dashboardVisible', isDashboardVisible);
}

// Restaurar estado do dashboard
function restoreDashboardState() {
  const savedState = localStorage.getItem('dashboardVisible');
  if (savedState === 'false') {
    toggleDashboard();
  }
}

// Executar na inicialização
document.addEventListener('DOMContentLoaded', restoreDashboardState);
