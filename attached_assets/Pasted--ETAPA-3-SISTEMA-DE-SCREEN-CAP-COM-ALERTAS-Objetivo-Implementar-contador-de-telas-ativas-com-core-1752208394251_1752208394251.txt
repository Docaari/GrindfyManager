JavaScript para Screen Cap
javascript// ===== SISTEMA DE SCREEN CAP =====
let screenCount = 0;
let screenCap = 5; // Valor do cap definido nos objetivos

function updateScreenCap() {
  const screenCapCard = document.querySelector('.screen-cap');
  const valueElement = screenCapCard.querySelector('.metric-value');
  const subElement = screenCapCard.querySelector('.metric-sub');
  
  const percentage = (screenCount / screenCap) * 100;
  
  valueElement.textContent = `${screenCount}/${screenCap}`;
  subElement.textContent = `${percentage.toFixed(0)}% do limite`;
  
  // Remover classes anteriores
  screenCapCard.classList.remove('warning', 'danger');
  
  // Aplicar cores de alerta
  if (percentage >= 100) {
    screenCapCard.classList.add('danger');
    showScreenCapAlert('danger');
  } else if (percentage >= 80) {
    screenCapCard.classList.add('danger');
  } else if (percentage >= 60) {
    screenCapCard.classList.add('warning');
  }
}

function incrementScreenCount() {
  screenCount++;
  updateScreenCap();
  updateMetrics();
}

function decrementScreenCount() {
  if (screenCount > 0) {
    screenCount--;
    updateScreenCap();
    updateMetrics();
  }
}

function showScreenCapAlert(type) {
  if (type === 'danger') {
    // Mostrar notificação de limite atingido
    showNotification('⚠️ Limite de telas atingido!', 'warning');
  }
}

function showNotification(message, type = 'info') {
  // Implementar sistema de notificações toast
  const notification = document.createElement('div');
  notification.className = `notification notification-${type}`;
  notification.textContent = message;
  
  document.body.appendChild(notification);
  
  setTimeout(() => {
    notification.remove();
  }, 5000);
}

// Integrar com ações de torneios
