2.2 Validação em Tempo Real
tsxconst validateMetrics = (field: string, value: number): boolean => {
  const validations = {
    volume: value >= 0,
    profit: !isNaN(value),
    abiMed: value >= 0,
    roi: !isNaN(value),
    fts: value >= 0 && value <= editData.volume,
    cravadas: value >= 0 && value <= editData.fts
  };
  
  return validations[field as keyof typeof validations] ?? true;
};
